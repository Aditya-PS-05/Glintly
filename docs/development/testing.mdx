## Overview

NovaCraft employs a comprehensive testing strategy covering unit tests, integration tests, and end-to-end tests. The testing stack includes Vitest for unit testing, React Testing Library for component testing, and Playwright for E2E testing.

### Vitest Configuration

The main test configuration is in `vitest.config.ts`:

```typescript vitest.config.ts
import { defineConfig } from 'vitest/config';
import react from '@vitejs/plugin-react';
import path from 'path';

export default defineConfig({
  plugins: [react()],
  test: {
    globals: true,
    environment: 'jsdom',
    setupFiles: ['./tests/setup.ts'],
    include: ['src/**/*.{test,spec}.{js,ts,jsx,tsx}'],
    exclude: ['node_modules', 'dist', 'e2e'],
    coverage: {
      reporter: ['text', 'json', 'html'],
      exclude: [
        'node_modules/',
        'src/generated/',
        'tests/',
        '**/*.d.ts',
        '**/*.config.js',
        '**/*.config.ts',
      ],
    },
  },
  resolve: {
    alias: {
      '@': path.resolve(__dirname, './src'),
    },
  },
});
```

import { render, screen, fireEvent, waitFor } from '@testing-library/react';
import { describe, it, expect, vi } from 'vitest';
import { ProjectCard } from '@/components/ProjectCard';
import { trpc } from '@/trpc/client';

// Mock tRPC
vi.mock('@/trpc/client', () => ({
  trpc: {
    projects: {
      delete: {
        useMutation: vi.fn(),
      },
    },
  },
}));

const mockProject = {
  id: '1',
  name: 'Test Project',
  description: 'A test project',
  createdAt: new Date('2023-01-01'),
  updatedAt: new Date('2023-01-01'),
  _count: { messages: 5 },
};

describe('ProjectCard', () => {
  it('renders project information correctly', () => {
    render(<ProjectCard project={mockProject} />);
    
    expect(screen.getByText('Test Project')).toBeInTheDocument();
    expect(screen.getByText('A test project')).toBeInTheDocument();
    expect(screen.getByText('5 messages')).toBeInTheDocument();
  });
  
  it('handles delete action', async () => {
    const mockDelete = vi.fn();
    const mockMutate = vi.fn();
    
    (trpc.projects.delete.useMutation as any).mockReturnValue({
      mutate: mockMutate,
      isLoading: false,
    });
    
    render(<ProjectCard project={mockProject} onDelete={mockDelete} />);
    
    const deleteButton = screen.getByLabelText('Delete project');
    fireEvent.click(deleteButton);
    
    // Confirm deletion
    const confirmButton = screen.getByText('Delete');
    fireEvent.click(confirmButton);
    
    await waitFor(() => {
      expect(mockMutate).toHaveBeenCalledWith({ id: '1' });
    });
  });
  
  it('shows loading state during deletion', () => {
    (trpc.projects.delete.useMutation as any).mockReturnValue({
      mutate: vi.fn(),
      isLoading: true,
    });
    
    render(<ProjectCard project={mockProject} />);
    
    expect(screen.getByText('Deleting...')).toBeInTheDocument();
  });
});

import { renderHook, waitFor } from '@testing-library/react';
import { describe, it, expect, vi } from 'vitest';
import { useProjectMessages } from '@/hooks/useProjectMessages';
import { trpc } from '@/trpc/client';

vi.mock('@/trpc/client');

describe('useProjectMessages', () => {

      {
        id: '1',
        content: 'Hello',
        role: 'USER',
        type: 'RESULT',
        createdAt: new Date(),
        fragments: [],
      },

      isLoading: false,
      error: null,
    });
    
    const { result } = renderHook(() => useProjectMessages('project-1'));
    
    await waitFor(() => {
      expect(result.current.messages).toEqual(mockMessages);
      expect(result.current.isLoading).toBe(false);
    });
  });

    const mockError = new Error('Failed to fetch messages');

      data: null,
      isLoading: false,
      error: mockError,
    });
    
    const { result } = renderHook(() => useProjectMessages('project-1'));
    
    await waitFor(() => {
      expect(result.current.error).toBe(mockError);
      expect(result.current.messages).toEqual([]);
    });
  });
});

```typescript tRPC Router Tests
import { describe, it, expect, beforeEach } from 'vitest';
import { createContext } from '@/trpc/context';
import { appRouter } from '@/trpc/routers/_app';
import { prisma } from '@/lib/prisma';
import { TRPCError } from '@trpc/server';

describe('Projects Router', () => {
  let ctx: Awaited<ReturnType<typeof createContext>>;
  let caller: ReturnType<typeof appRouter.createCaller>;
  
  beforeEach(async () => {
    ctx = await createContext({
      req: {} as any,
      res: {} as any,
    });
    caller = appRouter.createCaller(ctx);
  });
  
  it('creates a project successfully', async () => {
    const input = {
      name: 'Test Project',
      description: 'A test project'
    };
    
    const result = await caller.projects.create(input);
    
    expect(result.name).toBe(input.name);
    expect(result.description).toBe(input.description);
    expect(result.id).toBeDefined();
  });
  
  it('gets all projects', async () => {
    // Create test projects
    await prisma.project.createMany({
      data: [
        { name: 'Project 1', description: 'Description 1' },
        { name: 'Project 2', description: 'Description 2' },
      ]
    });
    
    const result = await caller.projects.getAll();
    
    expect(result).toHaveLength(2);
    expect(result[0].name).toBe('Project 1');
    expect(result[1].name).toBe('Project 2');
  });
  
  it('throws error for non-existent project', async () => {
    await expect(
      caller.projects.getById({ id: 'non-existent-id' })
    ).rejects.toThrow(TRPCError);
  });
  
  it('deletes project and cascades to messages', async () => {
    // Create project with messages
    const project = await prisma.project.create({
      data: {
        name: 'Test Project',
        messages: {
          create: [
            { content: 'Hello', role: 'USER', type: 'RESULT' },
            { content: 'Hi there', role: 'ASSISTANT', type: 'RESULT' }
          ]
        }
      }
    });
    
    await caller.projects.delete({ id: project.id });
    
    // Verify project is deleted
    const deletedProject = await prisma.project.findUnique({
      where: { id: project.id }
    });
    expect(deletedProject).toBeNull();
    
    // Verify messages are also deleted (cascade)
    const messages = await prisma.message.findMany({
      where: { projectId: project.id }
    });
    expect(messages).toHaveLength(0);
  });
});
```

import { test, expect } from '@playwright/test';

test.describe('Performance', () => {
  test('page load performance', async ({ page }) => {
    const response = await page.goto('/');
    
    // Verify response time
    expect(response?.status()).toBe(200);
    
    // Measure core web vitals
    const vitals = await page.evaluate(() => {
      return new Promise((resolve) => {
        new PerformanceObserver((list) => {
          const entries = list.getEntries();
          const vitals = {};
          
          entries.forEach((entry) => {
            if (entry.name === 'FCP') {
              vitals.fcp = entry.value;
            } else if (entry.name === 'LCP') {
              vitals.lcp = entry.value;
            } else if (entry.name === 'FID') {
              vitals.fid = entry.value;
            }
          });
          
          resolve(vitals);
        }).observe({ entryTypes: ['measure'] });
      });
    });
    
    // Assert performance metrics
    expect(vitals.fcp).toBeLessThan(2000); // FCP < 2s
    expect(vitals.lcp).toBeLessThan(2500); // LCP < 2.5s
  });
  
  test('message sending performance', async ({ page }) => {
    await page.goto('/projects/test-project-1');
    
    // Start performance measurement
    await page.evaluate(() => performance.mark('message-start'));
    
    // Send message
    await page.fill('[data-testid="message-input"]', 'Quick test message');
    await page.click('[data-testid="send-message"]');
    
    // Wait for response
    await expect(page.locator('[data-testid="message"][data-role="assistant"]')).toBeVisible();
    
    // End performance measurement
    const duration = await page.evaluate(() => {
      performance.mark('message-end');
      performance.measure('message-duration', 'message-start', 'message-end');
      const measure = performance.getEntriesByName('message-duration')[0];
      return measure.duration;
    });
    
    // Assert response time
    expect(duration).toBeLessThan(30000); // Response < 30s
  });
});

## Best Practices

### Test Organization

1. **File Structure**: Keep tests close to source code
2. **Naming**: Use descriptive test names
3. **Isolation**: Each test should be independent
4. **Mocking**: Mock external dependencies
5. **Performance**: Keep tests fast and focused

### Test Data

1. **Factories**: Use factories for creating test data
2. **Cleanup**: Clean up after tests
3. **Isolation**: Don't share state between tests
4. **Seeding**: Use consistent seed data for E2E tests

### Error Handling

1. **Expected Errors**: Test error scenarios
2. **Async Errors**: Handle async error testing
3. **Network Errors**: Mock network failures
4. **Database Errors**: Test database constraints
