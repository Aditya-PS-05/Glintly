## tRPC Configuration

### Server Setup

The main tRPC configuration is located in `src/trpc/trpc.ts`:

```typescript src/trpc/trpc.ts
import { TRPCError, initTRPC } from '@trpc/server';
import { Context } from './context';
import superjson from 'superjson';

const t = initTRPC.context<Context>().create({
  transformer: superjson,
  errorFormatter({ shape, error }) {
    return {
      ...shape,
      data: {
        ...shape.data,
        zodError: error.cause instanceof ZodError ? error.cause.flatten() : null,
      },
    };
  },
});

export const router = t.router;
export const publicProcedure = t.procedure;
export const protectedProcedure = t.procedure.use(
  t.middleware(({ ctx, next }) => {
    if (!ctx.session || !ctx.session.user) {
      throw new TRPCError({ code: 'UNAUTHORIZED' });
    }
    return next({
      ctx: {
        ...ctx,
        session: { ...ctx.session, user: ctx.session.user },
      },
    });
  })
);
```

### Context Provider

```typescript src/trpc/context.ts
import { inferAsyncReturnType } from '@trpc/server';
import { CreateNextContextOptions } from '@trpc/server/adapters/next';
import { prisma } from '@/lib/prisma';
import { getServerSession } from 'next-auth';
import { authOptions } from '@/lib/auth';

export async function createContext(opts: CreateNextContextOptions) {
  const session = await getServerSession(opts.req, opts.res, authOptions);
  
  return {
    session,
    prisma,
    req: opts.req,
    res: opts.res,
  };
}

export type Context = inferAsyncReturnType<typeof createContext>;
```

### Client Setup

```typescript src/trpc/client.ts
import { createTRPCNext } from '@trpc/next';
import { httpBatchLink, loggerLink, wsLink, splitLink } from '@trpc/client';
import { getUrl, transformer } from './shared';
import type { AppRouter } from './routers/_app';

function getEndingLink(ctx: any) {
  if (typeof window === 'undefined') {
    return httpBatchLink({
      url: getUrl(),
      headers() {
        return {
          cookie: ctx?.req?.headers?.cookie,
        };
      },
    });
  }
  
  return splitLink({
    condition(op) {
      return op.type === 'subscription';
    },
    true: wsLink({
      url: getUrl().replace(/^http/, 'ws'),
    }),
    false: httpBatchLink({
      url: getUrl(),
    }),
  });
}

export const trpc = createTRPCNext<AppRouter>({
  config({ ctx }) {
    return {
      transformer,
      links: [
        loggerLink({
          enabled: (opts) =>
            process.env.NODE_ENV === 'development' ||
            (opts.direction === 'down' && opts.result instanceof Error),
        }),
        getEndingLink(ctx),
      ],
    };
  },
  ssr: false,
});
```

### Projects Router

```typescript src/trpc/routers/projects.ts
import { router, publicProcedure } from '../trpc';
import { z } from 'zod';
import { TRPCError } from '@trpc/server';

export const projectsRouter = router({
  // Get all projects
  getAll: publicProcedure
    .query(async ({ ctx }) => {
      return await ctx.prisma.project.findMany({
        orderBy: { updatedAt: 'desc' },
        include: {
          _count: {
            select: { messages: true }
          }
        }
      });
    }),
  
  // Get single project with messages
  getById: publicProcedure
    .input(z.object({ id: z.string().cuid() }))
    .query(async ({ ctx, input }) => {
      const project = await ctx.prisma.project.findUnique({
        where: { id: input.id },
        include: {
          messages: {
            include: {
              fragments: true
            },
            orderBy: { createdAt: 'asc' }
          }
        }
      });
      
      if (!project) {
        throw new TRPCError({
          code: 'NOT_FOUND',
          message: 'Project not found'
        });
      }
      
      return project;
    }),
  
  // Create new project
  create: publicProcedure
    .input(z.object({
      name: z.string().min(1).max(100),
      description: z.string().optional()
    }))
    .mutation(async ({ ctx, input }) => {
      return await ctx.prisma.project.create({
        data: {
          name: input.name,
          description: input.description
        }
      });
    }),
  
  // Update project
  update: publicProcedure
    .input(z.object({
      id: z.string().cuid(),
      name: z.string().min(1).max(100).optional(),
      description: z.string().optional()
    }))
    .mutation(async ({ ctx, input }) => {
      const { id, ...data } = input;
      
      return await ctx.prisma.project.update({
        where: { id },
        data: data
      });
    }),
  
  // Delete project
  delete: publicProcedure
    .input(z.object({ id: z.string().cuid() }))
    .mutation(async ({ ctx, input }) => {
      return await ctx.prisma.project.delete({
        where: { id: input.id }
      });
    }),
  
  // Get project statistics
  getStats: publicProcedure
    .query(async ({ ctx }) => {
      const [totalProjects, totalMessages, totalFragments] = await Promise.all([
        ctx.prisma.project.count(),
        ctx.prisma.message.count(),
        ctx.prisma.fragment.count()
      ]);
      
      return {
        totalProjects,
        totalMessages,
        totalFragments,
        averageMessagesPerProject: totalMessages / Math.max(totalProjects, 1)
      };
    })
});
```

### Messages Router

```typescript src/trpc/routers/messages.ts
import { router, publicProcedure } from '../trpc';
import { z } from 'zod';
import { observable } from '@trpc/server/observable';
import { EventEmitter } from 'events';

// Event emitter for real-time updates
const messageEvents = new EventEmitter();

export const messagesRouter = router({
  // Get messages for a project
  getByProject: publicProcedure
    .input(z.object({ 
      projectId: z.string().cuid(),
      cursor: z.string().optional(),
      limit: z.number().min(1).max(100).default(50)
    }))
    .query(async ({ ctx, input }) => {
      const messages = await ctx.prisma.message.findMany({
        where: { projectId: input.projectId },
        cursor: input.cursor ? { id: input.cursor } : undefined,
        take: input.limit + 1, // +1 to check if there's a next page
        orderBy: { createdAt: 'asc' },
        include: {
          fragments: true
        }
      });
      
      let nextCursor: string | undefined = undefined;
      if (messages.length > input.limit) {
        const nextItem = messages.pop();
        nextCursor = nextItem!.id;
      }
      
      return {
        messages,
        nextCursor
      };
    }),
  
  // Create new message
  create: publicProcedure
    .input(z.object({
      projectId: z.string().cuid(),
      content: z.string().min(1),
      role: z.enum(['USER', 'ASSISTANT']),
      type: z.enum(['RESULT', 'ERROR']).default('RESULT')
    }))
    .mutation(async ({ ctx, input }) => {
      const message = await ctx.prisma.message.create({
        data: input,
        include: {
          fragments: true
        }
      });
      
      // Emit real-time event
      messageEvents.emit('messageCreated', {
        projectId: input.projectId,
        message
      });
      
      return message;
    }),
  
  // Update message
  update: publicProcedure
    .input(z.object({
      id: z.string().cuid(),
      content: z.string().min(1).optional(),
      type: z.enum(['RESULT', 'ERROR']).optional()
    }))
    .mutation(async ({ ctx, input }) => {
      const { id, ...data } = input;
      
      const message = await ctx.prisma.message.update({
        where: { id },
        data: data,
        include: {
          fragments: true
        }
      });
      
      // Emit real-time event
      messageEvents.emit('messageUpdated', {
        projectId: message.projectId,
        message
      });
      
      return message;
    }),
  
  // Delete message
  delete: publicProcedure
    .input(z.object({ id: z.string().cuid() }))
    .mutation(async ({ ctx, input }) => {
      const message = await ctx.prisma.message.delete({
        where: { id: input.id }
      });
      
      // Emit real-time event
      messageEvents.emit('messageDeleted', {
        projectId: message.projectId,
        messageId: message.id
      });
      
      return message;
    }),
  
  // Real-time subscription for project messages
  onMessageChange: publicProcedure
    .input(z.object({ projectId: z.string().cuid() }))
    .subscription(({ input }) => {
      return observable<{
        type: 'created' | 'updated' | 'deleted';
        message?: any;
        messageId?: string;
      }>((emit) => {
        const onCreated = (data: any) => {
          if (data.projectId === input.projectId) {
            emit.next({ type: 'created', message: data.message });
          }
        };
        
        const onUpdated = (data: any) => {
          if (data.projectId === input.projectId) {
            emit.next({ type: 'updated', message: data.message });
          }
        };
        
        const onDeleted = (data: any) => {
          if (data.projectId === input.projectId) {
            emit.next({ type: 'deleted', messageId: data.messageId });
          }
        };
        
        messageEvents.on('messageCreated', onCreated);
        messageEvents.on('messageUpdated', onUpdated);
        messageEvents.on('messageDeleted', onDeleted);
        
        return () => {
          messageEvents.off('messageCreated', onCreated);
          messageEvents.off('messageUpdated', onUpdated);
          messageEvents.off('messageDeleted', onDeleted);
        };
      });
    })
});
```

## Error Handling

### Server-Side Error Handling

```typescript Error Handling
import { TRPCError } from '@trpc/server';

export const projectsRouter = router({
  getById: publicProcedure
    .input(z.object({ id: z.string().cuid() }))
    .query(async ({ ctx, input }) => {
      try {
        const project = await ctx.prisma.project.findUnique({
          where: { id: input.id },
          include: { messages: true }
        });
        
        if (!project) {
          throw new TRPCError({
            code: 'NOT_FOUND',
            message: 'Project not found',
            cause: new Error(`Project ${input.id} does not exist`)
          });
        }
        
        return project;
      } catch (error) {
        if (error instanceof TRPCError) {
          throw error;
        }
        
        // Handle unexpected errors
        throw new TRPCError({
          code: 'INTERNAL_SERVER_ERROR',
          message: 'An unexpected error occurred',
          cause: error
        });
      }
    })
});
```

## Real-time Features

### WebSocket Subscriptions

```typescript Real-time Usage
function ProjectMessages({ projectId }: { projectId: string }) {
  const [messages, setMessages] = useState<Message[]>([]);
  
  // Subscribe to real-time updates
  trpc.messages.onMessageChange.useSubscription(
    { projectId },
    {
      onData: (data) => {
        switch (data.type) {
          case 'created':
            setMessages(prev => [...prev, data.message]);
            break;
          case 'updated':
            setMessages(prev => prev.map(msg => 
              msg.id === data.message.id ? data.message : msg
            ));
            break;
          case 'deleted':
            setMessages(prev => prev.filter(msg => msg.id !== data.messageId));
            break;
        }
      },
      onError: (error) => {
        console.error('Subscription error:', error);
      }
    }
  );
  
  return (
    <div>
      {messages.map(message => (
        <div key={message.id}>
          <strong>{message.role}:</strong> {message.content}
        </div>
      ))}
    </div>
  );
}
```

```typescript Optimistic Updates
function CreateMessageForm({ projectId }: { projectId: string }) {
  const utils = trpc.useContext();
  
  const createMessage = trpc.messages.create.useMutation({
    onMutate: async (newMessage) => {
      // Cancel outgoing queries
      await utils.messages.getByProject.cancel({ projectId });
      
      // Snapshot previous value
      const previousMessages = utils.messages.getByProject.getData({ projectId });
      
      // Optimistically update
      utils.messages.getByProject.setData({ projectId }, (old) => {
        if (!old) return old;
        return {
          ...old,
          messages: [...old.messages, {
            id: 'temp-id',
            ...newMessage,
            createdAt: new Date(),
            fragments: []
          }]
        };
      });
      
      return { previousMessages };
    },
    onError: (err, newMessage, context) => {
      // Revert optimistic update on error
      utils.messages.getByProject.setData(
        { projectId },
        context?.previousMessages
      );
    },
    onSettled: () => {
      // Refetch after mutation
      utils.messages.getByProject.invalidate({ projectId });
    }
  });
  
  return (
    <form onSubmit={(e) => {
      e.preventDefault();
      createMessage.mutate({
        projectId,
        content: formData.content,
        role: 'USER'
      });
    }}>
      {/* Form fields */}
    </form>
  );
}
```
