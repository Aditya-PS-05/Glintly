## Overview

The Messages API handles all communication between users and AI agents. It manages message creation, retrieval, and the triggering of AI processing through Inngest jobs.

## Endpoints

### Send Message

Create a new message and trigger AI processing.

<CodeGroup>

```typescript tRPC Mutation
const sendMessage = trpc.messages.send.useMutation({
  onSuccess: (message) => {
    console.log('Message sent:', message);
  }
});

sendMessage.mutate({
  projectId: "clx1234567890abcdef",
  content: "Create a todo application with React",
  role: "USER",
  type: "RESULT"
});
```

```typescript Server Call
const message = await trpc.messages.send.mutate({
  projectId: "clx1234567890abcdef",
  content: "Create a todo application with React",
  role: "USER",
  type: "RESULT"
});
```

</CodeGroup>

**Parameters**

<ResponseField name="projectId" type="string" required>
  The unique identifier of the project
</ResponseField>

<ResponseField name="content" type="string" required>
  The message content (1-10,000 characters)
</ResponseField>

<ResponseField name="role" type="'USER' | 'ASSISTANT'" required>
  The role of the message sender
</ResponseField>

<ResponseField name="type" type="'RESULT' | 'ERROR'" required>
  The type of message
</ResponseField>

**Response**

<CodeGroup>



## Message Processing Flow

### User Message Processing

When a USER message is sent, the following flow is triggered:

1. **Message Validation**: Input is validated using Zod schema
2. **Database Storage**: Message is stored in the database
3. **Inngest Job**: `code-agent/run` event is triggered
4. **AI Processing**: AI agent processes the message and generates code
5. **Response Creation**: Assistant message is created with results
6. **Fragment Storage**: Code fragment is stored if successful

<CodeGroup>

```typescript Processing Flow
// 1. User sends message
const userMessage = await trpc.messages.send.mutate({
  projectId: "project-123",
  content: "Create a React todo app",
  role: "USER",
  type: "RESULT"
});

// 2. Inngest job is triggered automatically
// 3. AI processes the request
// 4. Assistant message is created
const assistantMessage = {
  id: "msg-456",
  content: "I've created a todo application...",
  role: "ASSISTANT",
  type: "RESULT",
  fragment: {
    sandboxUrl: "https://sandbox-123.e2b.dev",
    title: "Todo Application",
    files: { /* generated files */ }
  }
};
```

</CodeGroup>

## Usage Examples

### Chat Interface

<CodeGroup>

```typescript Chat Component
import { useState } from 'react';
import { trpc } from '@/trpc/client';

export function ChatInterface({ projectId }: { projectId: string }) {
  const [input, setInput] = useState('');
  const [isLoading, setIsLoading] = useState(false);
  
  const { data: messages } = trpc.messages.getByProject.useQuery(
    { projectId },
    { refetchInterval: 2000 }
  );
  
  const sendMessage = trpc.messages.send.useMutation({
    onMutate: () => setIsLoading(true),
    onSettled: () => setIsLoading(false),
    onError: (error) => {
      console.error('Failed to send message:', error);
    }
  });
  
  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    if (!input.trim()) return;
    
    sendMessage.mutate({
      projectId,
      content: input,
      role: 'USER',
      type: 'RESULT'
    });
    
    setInput('');
  };
  
  return (
    <div className="flex flex-col h-full">
      <div className="flex-1 overflow-y-auto">
        {messages?.map(message => (
          <div key={message.id} className={`message ${message.role}`}>
            <div className="content">{message.content}</div>
            {message.fragment && (
              <div className="fragment">
                <a href={message.fragment.sandboxUrl} target="_blank">
                  View {message.fragment.title}
                </a>
              </div>
            )}
          </div>
        ))}
      </div>
      
      <form onSubmit={handleSubmit} className="border-t p-4">
        <div className="flex gap-2">
          <input
            type="text"
            value={input}
            onChange={(e) => setInput(e.target.value)}
            placeholder="Describe what you want to build..."
            className="flex-1 px-3 py-2 border rounded"
            disabled={isLoading}
          />
          <button
            type="submit"
            disabled={isLoading || !input.trim()}
            className="px-4 py-2 bg-blue-500 text-white rounded disabled:opacity-50"
          >
            {isLoading ? 'Sending...' : 'Send'}
          </button>
        </div>
      </form>
    </div>
  );
}
```

### Optimistic Updates

<CodeGroup>

```typescript Optimistic Updates
export function useOptimisticMessages(projectId: string) {
  const utils = trpc.useUtils();
  
  const sendMessage = trpc.messages.send.useMutation({
    onMutate: async (newMessage) => {
      // Cancel any outgoing refetches
      await utils.messages.getByProject.cancel({ projectId });
      
      // Snapshot the previous value
      const previousMessages = utils.messages.getByProject.getData({ projectId });
      
      // Optimistically update to the new value
      const optimisticMessage = {
        id: 'temp-' + Date.now(),
        ...newMessage,
        createdAt: new Date(),
        updatedAt: new Date(),
        fragment: null
      };
      
      utils.messages.getByProject.setData({ projectId }, old => [
        ...(old || []),
        optimisticMessage
      ]);
      
      return { previousMessages };
    },
    
    onError: (err, newMessage, context) => {
      // Rollback on error
      utils.messages.getByProject.setData(
        { projectId },
        context?.previousMessages
      );
    },
    
    onSettled: () => {
      // Always refetch after error or success
      utils.messages.getByProject.invalidate({ projectId });
    }
  });
  
  return { sendMessage };
}
```

</CodeGroup>

### Error Recovery

<CodeGroup>

```typescript Error Recovery
export function useMessageWithRetry(projectId: string) {
  const [retryCount, setRetryCount] = useState(0);
  const maxRetries = 3;
  
  const sendMessage = trpc.messages.send.useMutation({
    onError: (error) => {
      if (retryCount < maxRetries) {
        setTimeout(() => {
          setRetryCount(prev => prev + 1);
          sendMessage.mutate(/* same parameters */);
        }, 1000 * Math.pow(2, retryCount)); // Exponential backoff
      }
    },
    onSuccess: () => {
      setRetryCount(0);
    }
  });
  
  return { sendMessage, retryCount, canRetry: retryCount < maxRetries };
}
```

</CodeGroup>
